\part a

\align{
P(X\ge1) &= P(X=1) + P(X=2) + P(X=3) \\
&= 0.21 + 0.40 + 0.24 \\
&= 0.85
}

\part b

\align{
E(X)
&= \sum_{n=0}^3 nP(X=n) \\
&= 0\*0.15 + 1\*0.21 + 2\*0.40 + 3\*0.24 \\
&= 1.73
}

\part c

\align{
P(X=3|X\ge1) &= \frac{P(X=3)}{P(X\ge1)} \\
&= \frac{0.24}{0.85} \\
&= 0.2824
}

\part d

\align{
E(X|X\ge1)
&= \sum_{n=0}^3 nP(X=n|X\ge1) \\
&= \frac{\displaystyle\sum_{n=1}^3 nP(X=n)}{P(X\ge1)} \\
&= \frac{1\*0.21 + 2\*0.40 + 3\*0.24}{0.85} \\
&= 2.035
}\\[1mm]

\therefore E(X|X\ge1) > E(X)
